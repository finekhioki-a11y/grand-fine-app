export default { async fetch(request, env, ctx) { const url = 'https://www.hotenavi.com/fine017/empty'; const response = await fetch(url); const text = await response.text(); // ページ内の「空室：〇室」という部分を確実に拾うための修正 const regex = /空室：(\d+)室/; const match = text.match(regex); const count = match ? match[1] : "0"; // 見つからなければ0とする return new Response(JSON.stringify({ count: count }), { headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } }); } };
